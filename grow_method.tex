设 terminal percentage 为 $t \in [0,1]$，树有二叉。

\begin{equation}
    \begin{aligned}
        P(D=d) = 
        \begin{cases}
            t,  d=1 \\
            % P(D \leq d-1)\cdot P(D=d-1),  d>1 \\
            (1-t) \cdot [(P(D = d-1))^2 + 2 \cdot P(D = d-1) \cdot P(D < d-1)],  d \geq 2 \\
        \end{cases}
    \end{aligned}
\end{equation}

\begin{equation}
    \begin{aligned}
        P(N=n) = 
        \begin{cases}
            t,  n=1 \\
            (1-t) \cdot \sum_{l=1}^{n-2} P(N=l) \cdot P(N=n-1-l),  d \geq 2 \\
        \end{cases}
    \end{aligned}
\end{equation}

则可求期望
\begin{equation}
    \begin{aligned}
        E(D) = \sum_{d=1}^{\infty} d \cdot P(D=d) \\
        E(N) = \sum_{n=1}^{\infty} n \cdot P(N=n) \\
    \end{aligned}
\end{equation}
